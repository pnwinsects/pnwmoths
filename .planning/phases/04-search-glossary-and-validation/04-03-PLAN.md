1. Create `scripts/check-page-weight.js` as a Node.js ESM script consistent with the `scripts/build-data.js` pattern. Threshold 500KB, warn-only (per D-11):

```javascript
// scripts/check-page-weight.js
// Post-build check: warns when any HTML page exceeds the configured threshold.
// Run via: npm run build:check-weight
import { readdirSync, statSync } from 'node:fs';
import { join } from 'node:path';

const THRESHOLD_BYTES = 500 * 1024; // 500KB per D-11
const SITE_DIR = '_site';

let warnCount = 0;

function walkHtml(dir) {
  const entries = readdirSync(dir, { withFileTypes: true });
  for (const entry of entries) {
    const fullPath = join(dir, entry.name);
    if (entry.isDirectory()) {
      walkHtml(fullPath);
    } else if (entry.name.endsWith('.html')) {
      const { size } = statSync(fullPath);
      if (size > THRESHOLD_BYTES) {
        console.warn(`[page-weight] WARNING: ${fullPath} is ${(size / 1024).toFixed(1)}KB (threshold: 500KB)`);
        warnCount++;
      }
    }
  }
}

walkHtml(SITE_DIR);
if (warnCount > 0) {
  console.warn(`[page-weight] ${warnCount} page(s) exceed 500KB threshold.`);
} else {
  console.log('[page-weight] All pages under 500KB threshold.');
}
```

Note: This script exits 0 even when warnings are emitted (warn-only per D-11). It does NOT abort the build.

```json

    "build:copy-parquet": "node scripts/copy-parquet.js",
    "build:pagefind": "pagefind --site _site",
    "build:validate-links": "lychee --root-dir _site --offline '_site/**/*.html'",
    "build:check-weight": "node scripts/check-page-weight.js",
    "build": "npm run build:data && npm run build:eleventy && npm run build:copy-parquet && npm run build:pagefind && npm run build:validate-links && npm run build:check-weight",

}
```

Key details:
- `build:pagefind` uses `pagefind` (not `npx pagefind`) since it is now a devDependency, npm resolves it from node_modules/.bin/
- `build:pagefind` must run AFTER `build:eleventy` (it indexes the built HTML) and AFTER `build:copy-parquet`
- `build:validate-links` uses `--offline` flag to skip external URLs, VALD-01 only requires checking internal links (per D-10). `--root-dir _site` resolves root-relative URLs like /species/acronicta-americana/
- `build:check-weight` runs last (it is warn-only and should not block subsequent steps if we add more later)
- The full `build` chain order: data -> eleventy -> copy-parquet -> pagefind -> validate-links -> check-weight
