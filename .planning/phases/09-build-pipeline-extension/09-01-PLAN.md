    ```javascript
    // scripts/emit-species-states.js
    // Post-Vite build step: query records.csv via DuckDB, write _site/species-states.json
    // Run via: npm run build:species-states
    import { DuckDBInstance } from '@duckdb/node-api';
    import { writeFileSync, mkdirSync } from 'node:fs';
    import { resolve } from 'node:path';

    export async function main() {
      const db = await DuckDBInstance.create(':memory:');
      const conn = await db.connect();

      await conn.run(`
        CREATE TABLE records AS
        SELECT * FROM read_csv('data/records.csv',
          header = true,
          columns = {
            'species_slug': 'VARCHAR',
            'record_type': 'VARCHAR',
            'latitude': 'DOUBLE',
            'longitude': 'DOUBLE',
            'state': 'VARCHAR',
            'county': 'VARCHAR',
            'locality': 'VARCHAR',
            'elevation_ft': 'INTEGER',
            'year': 'INTEGER',
            'month': 'INTEGER',
            'day': 'INTEGER',
            'collector': 'VARCHAR',
            'collection': 'VARCHAR',
            'notes': 'VARCHAR'
          }
        )
      `);

      const result = await conn.runAndReadAll(`
        SELECT DISTINCT species_slug, state
        FROM records
        WHERE state IS NOT NULL AND state != ''
        ORDER BY species_slug, state
      `);
      const rows = result.getRowObjectsJS();

      conn.closeSync();

      const outPath = resolve('_site/species-states.json');
      mkdirSync(resolve('_site'), { recursive: true });
      writeFileSync(outPath, JSON.stringify(rows));
      console.log(`Wrote ${rows.length} species-state pairs to _site/species-states.json`);
    }

    if (import.meta.url === `file://${process.argv[1]}`) {
      main().catch(err => {
        console.error(err.message);
        process.exit(1);
      });
    }
    ```

| Boundary | Description |

| DuckDB → _site/species-states.json | Build-time write to output directory; no runtime trust boundary |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09-01 | Tampering | `species_slug` values in records.csv written verbatim to JSON | mitigate | `validateSlugComponent` already enforced in `build-data.js` for all species_slug values before Parquet export; same slug values flow into `_site/species-states.json`. The SELECT DISTINCT query reads from `records.csv` which is validated by `build-data.js` first (build chain runs `build:data` before `build:species-states`). Residual risk: slug not re-validated in emit-species-states.js directly, but build chain ordering ensures pre-validation. |
| T-09-02 | Denial of Service | Unbounded output file size | accept | `SELECT DISTINCT` caps output at (unique species × unique states). With ~700 species × 6 known states = max 4,200 rows × ~40 bytes = ~168 KB. Validated state allowlist in build-data.js further bounds states. Risk accepted: file size is deterministically bounded by data constraints. |
| T-09-03 | Information Disclosure | `_site/species-states.json` exposed publicly | accept | File contains only species_slug and state name — both already public on the site. No PII, coordinates, or sensitive data included. Risk accepted. |
